 38     if (typeof DOMParser == "undefined") {
 39         DOMParser = function() {

 41         DOMParser.prototype.parseFromString = function(string, contentType) {
 42             if (typeof ActiveXObject != "undefined") {
 43                 var domDoc = new ActiveXObject("MSXML.DomDocument");
 44                 domDoc.loadXML(string);
 45                 return domDoc;
 46             } else if (typeof XMLHttpRequest != "undefined") {
 47                 var xhr = new XMLHttpRequest();
 48                 xhr.open("GET", "data:" + (contenttype || "application/xml")
 49                         + ";charset=utf-8," + encodingURIComponent(string),
 50                         false);
 51                 if (xhr.overrideMimeType) {
 52                     xhr.overrideMimeType(contentType);
 53                 }
 54                 xhr.send(null);
 55                 return xhr.responseXML;
 56             }
 57         };
 58     }

 60     /**
 61      * BulletMLを解析し、JavaScriptオブジェクトツリーを生成する.
 62      * 
 63      * @param {String|Document}
 64      *            xml
 65      * @return {BulletML.Root}
 66      */
 67     BulletML.build = function(xml) {
 68         var result;
 69         if (typeof (xml) == "string") {
 70             var domParser = new DOMParser();
 71             result = parse(domParser.parseFromString(xml, "application/xml"));
 72         } else if (xml.getElementsByTagName("bulletml")) {
 73             result = parse(xml);
 74         } else {
 75             throw new Exception("cannot build " + xml);
 76         }
 77 
 78         // find topAction
 79         result.topAction = search(result.actions, "top");
 80         return result;
 81     };

170     /**
171      * コマンドツリーをシーケンシャルにアクセス可能な配列に変換する.
172      * 
173      * 変換に際し、各種パラメータ内の変数はサブノード呼び出しパラメータ（param要素）が展開される.<br>
174      * ランク（$rank）、乱数（$rand）もこの時点で確定される.
175      * 
176      * @param {string}
177      *            label label attribute value
178      * @return {Array.<BulletML.Command>}
179      * @memberOf BulletML.Root.prototype
180      */
181     BulletML.Root.prototype.sequence = function(actionLabel) {
182         if (!actionLabel && !this.topAction) {
183             throw new Error("has no top action(s).");
184         }
185         var topAction;
186         if (actionLabel) {
187             topAction = this.findAction(actionLabel);
188         } else {
189             topAction = this.topAction;
190         }
191         var visitor = new BulletML.Visitor(this);
192         visitor.visit(topAction);
193         return visitor.result;
194     };

196     /**
197      * @constructor
198      */
199     BulletML.Visitor = function(root) {
200         /**
201          * @type {BulletML.Root}
202          * @field
203          */
204         this.root = root;
205         /**
206          * @type {Array.<BulletML.Command>}
207          * @field
208          */
209         this.result = [];
210         this.paramsStack = [];
211     };
212     /**
213      * @memberOf BulletML.Visitor.prototype
214      */
215     BulletML.Visitor.prototype.visit = function(command) {
216         switch (command.commandName) {
217         case "action":
218             for ( var i = 0, end = command.commands.length; i < end; i++) {
219                 this.visit(command.commands[i]);
220             }
221             break;
222         case "actionRef":
223             this.pushParams(command.params);
224             var action = this.root.findAction(command.label);
225             for ( var i = 0, end = action.commands.length; i < end; i++) {
226                 this.visit(action.commands[i]);
227             }
228             this.paramsStack.pop();
229             break;
230         case "repeat":
231             var start = new BulletML.LoopStart();
232             var times = evalNumber(command.times, this.params(), this.root.rank);
233             var end = new BulletML.LoopEnd(start, times);
234 
235             this.result.push(start);
236             this.visit(command.action);
237             this.result.push(end);
238             break;
239         default:
240             this.result.push(command.clone(this.params()));
241         }
242     };
243     /**
244      * @memberOf BulletML.Visitor.prototype
245      */
246     BulletML.Visitor.prototype.pushParams = function(params) {
247         var cp = this.params();
248         var result = [];
249         for ( var i = 0, end = params.length; i < end; i++) {
250             result.push(evalNumberFixRand(params[i], cp, this.root.rank))
251         }
252         this.paramsStack.push(result);
253     };
254     /**
255      * @memberOf BulletML.Visitor.prototype
256      */
257     BulletML.Visitor.prototype.params = function() {
258         if (this.paramsStack.length == 0) {
259             return [];
260         }
261         return this.paramsStack[this.paramsStack.length - 1];
262     };

269     BulletML.Bullet = function() {
270         /**
271          * @type {string}
272          * @field
273          */
274         this.label = null;
275         /**
276          * @type {BulletML.Root}
277          * @field
278          */
279         this.root = null;
280         /**
281          * @type {BulletML.Direction}
282          * @field
283          */
284         this.direction = new BulletML.Direction(0);
285         /**
286          * @type {BulletML.Speed}
287          * @field
288          */
289         this.speed = new BulletML.Speed(1);
290         /**
291          * @type {Array.<BulletML.Command>}
292          * @field
293          */
294         this.actions = [];
295     };
296     /**
297      * @memberOf BulletML.Bullet.prototype
298      */
299     BulletML.Bullet.prototype.sequence = function() {
300         var visitor = new BulletML.Visitor(this.root);
301         for ( var i = 0, end = this.actions.length; i < end; i++) {
302             visitor.visit(this.actions[i]);
303         }
304         return visitor.result;
305     };
306     /**
307      * @memberOf BulletML.Bullet.prototype
308      */
309     BulletML.Bullet.prototype.clone = function(params) {
310         var result = new BulletML.Bullet();
311         result.label = this.label;
312         result.root = this.root;
313         if (this.direction) {
314             result.direction = new BulletML.Direction(evalNumber(
315                     this.direction.value, params, this.root.rank));
316             result.direction.type = this.direction.type;
317         }
318         if (this.speed) {
319             result.speed = new BulletML.Speed(evalNumber(this.speed.value,
320                     params, this.root.rank));
321             result.speed.type = this.speed.type;
322         }
323         for ( var i = 0, end = this.actions.length; i < end; i++) {
324             result.actions.push(this.actions[i].clone(params));
325         }
326         return result;
327     };

479     BulletML.Fire.prototype = new BulletML.Command();
480     /**
481      * @memberOf BulletML.Fire.prototype
482      */
483     BulletML.Fire.prototype.clone = function(params) {
484         var result = new BulletML.Fire();
485         result.label = this.label;
486         result.root = this.root;
487         if (this.direction) {
488             result.direction = new BulletML.Direction(evalNumber(
489                     this.direction.value, params, this.root.rank));
490             result.direction.type = this.direction.type;
491         }
492         if (this.speed) {
493             result.speed = new BulletML.Speed(evalNumber(this.speed.value,
494                     params, this.root.rank));
495             result.speed.type = this.speed.type;
496         }
497 
498         if (this.bullet) {
499             if (this.bullet instanceof BulletML.Bullet) {
500                 result.bullet = this.bullet.clone(params);
501             } else if (this.bullet instanceof BulletML.BulletRef) {
502                 var origBullet = this.root.findBullet(this.bullet.label);
503                 if (!origBullet) {
504                     return result;
505                 }
506                 var newParam = [];
507                 for ( var i = 0, end = this.bullet.params.length; i < end; i++) {
508                     newParam.push(evalNumberFixRand(this.bullet.params[i],
509                             params, this.root.rank));
510                 }
511                 result.bullet = origBullet.clone(newParam);
512             }
513         }
514         return result;
515     }

853     function parse(element) {
854         var result = new BulletML.Root();
855 
856         var root = element.getElementsByTagName("bulletml")[0];
857         attr(root, "type", function(type) {
858             result.type = type;
859         });
860 
861         // Top Level Actions
862         var actions = root.getElementsByTagName("action");
863         if (actions) {
864             for ( var i = 0, end = actions.length; i < end; i++) {
865                 var newAction = parseAction(result, actions[i]);
866                 if (newAction) {
867                     result.actions.push(newAction);
868                 }
869             }
870         }
871 
872         // Top Level Bullets
873         var bullets = root.getElementsByTagName("bullet");
874         if (bullets) {
875             for ( var i = 0, end = bullets.length; i < end; i++) {
876                 var newBullet = parseBullet(result, bullets[i]);
877                 if (newBullet) {
878                     result.bullets.push(newBullet);
879                 }
880             }
881         }
882 
883         // Top Level Fires
884         var fires = root.getElementsByTagName("fire");
885         if (fires) {
886             for ( var i = 0, end = fires.length; i < end; i++) {
887                 var newFire = parseFire(result, fires[i]);
888                 if (newFire) {
889                     result.fires.push(newFire);
890                 }
891             }
892         }
893 
894         return result;
895     }

897     function parseAction(root, element) {
898         var result = new BulletML.Action();
899         attr(element, "label", function(label) {
900             result.label = label;
901         });
902         each(element, ".", function(commandElm) {
903             switch (commandElm.tagName) {
904             case "action":
905                 result.commands.push(parseAction(root, commandElm));
906                 break;
907             case "actionRef":
908                 result.commands.push(parseActionRef(root, commandElm));
909                 break;
910             case "fire":
911                 result.commands.push(parseFire(root, commandElm));
912                 break;
913             case "fireRef":
914                 result.commands.push(parseFireRef(root, commandElm));
915                 break;
916             case "changeDirection":
917                 result.commands.push(parseChangeDirection(root, commandElm));
918                 break;
919             case "changeSpeed":
920                 result.commands.push(parseChangeSpeed(root, commandElm));
921                 break;
922             case "accel":
923                 result.commands.push(parseAccel(root, commandElm));
924                 break;
925             case "wait":
926                 result.commands.push(parseWait(root, commandElm));
927                 break;
928             case "vanish":
929                 result.commands.push(parseVanish(root, commandElm));
930                 break;
931             case "repeat":
932                 result.commands.push(parseRepeat(root, commandElm));
933                 break;
934             }
935         });
936 
937         result.root = root;
938         return result;
939     }

1198     function attr(element, attrName, callback, ifNotFound) {
1199         // for IE
1200         var attrs;

1202             attrs = {};
1203             var i = 0;
1204             while (element.attributes.item(i)) {
1205                 var item = element.attributes.item(i);
1206                 if (item.value != "" && item.value != "null") {
1207                     attrs[item.name] = {
1208                         value : item.value
1209                     };
1210                 }
1211                 i++;
1212             }
1213         } else {
1214             attrs = element.attributes;
1215         }
1216 
1217         var attr = attrs[attrName];
1218         if (attr) {
1219             if (callback) {
1220                 callback(attr.value);
1221             }
1222             return attr;
1223         } else if (ifNotFound) {
1224             ifNotFound();
1225         }
