 16     /**
 17      * bulletmlのルート要素.
 18      *
 19      * @constructor
 20      * @param {Object=} data
 21      */
 22     bulletml.Root = function(data) {
 23         /**
 24          * @type {string}
 25          */
 26         this.type = "none";
 27         /**
 28          * @type {bulletml.Root}
 29          */
 30         this.root = this;
 31         /**
 32          * top level action elements.
 33          *
 34          * @type {Array.<bulletml.Action>}
 35          */
 36         this.actions = [];
 37         /**
 38          * top level bullet elements.
 39          *
 40          * @type {Array.<bulletml.Bullet>}
 41          */
 42         this.bullets = [];
 43         /**
 44          * top level fire elements.
 45          *
 46          * @type {Array.<bulletml.Fire>}
 47          */
 48         this.fires = [];
 49 
 50         if (data !== undefined) {
 51             for (var prop in data) if (data.hasOwnProperty(prop)) {
 52                 data[prop].label = prop;
 53                 if (data[prop] instanceof bulletml.Action) {
 54                     this.actions.push(data[prop]);
 55                 } else if (data[prop] instanceof bulletml.Bullet) {
 56                     this.bullets.push(data[prop]);
 57                 } else if (data[prop] instanceof bulletml.Fire) {
 58                     this.fires.push(data[prop]);
 59                 }
 60             }
 61 
 62             for (var i = 0, end = this.actions.length; i < end; i++) {
 63                 this.actions[i].setRoot(this);
 64             }
 65             for (var i = 0, end = this.bullets.length; i < end; i++) {
 66                 this.bullets[i].setRoot(this);
 67             }
 68             for (var i = 0, end = this.fires.length; i < end; i++) {
 69                 this.fires[i].setRoot(this);
 70             }
 71         }
 72     };

 85     /**
 86      * find actions label starts with 'top'.
 87      *
 88      * @return Array.<bulletml.Action>
 89      */
 90     bulletml.Root.prototype.getTopActionLabels = function() {
 91         var result = [];
 92         for ( var i = 0, end = this.actions.length; i < end; i++) {
 93             var action = this.actions[i];
 94             if (action.label && action.label.indexOf("top") === 0) {
 95                 result[result.length] = action.label;
 96             }
 97         }
 98         return result;
 99     };

206     /**
207      * @return {bulletml.Bullet}
208      */
209     bulletml.Bullet.prototype.clone = function(walker) {
210         var c = new bulletml.Bullet();
211         c.label = this.label;
212         c.root = this.root;
213         c.actions = this.actions;
214         c.direction = new bulletml.Direction(walker._evalParam(this.direction.value));
215         c.direction.type = this.direction.type;
216         c.speed = new bulletml.Speed(walker._evalParam(this.speed.value));
217         c.speed.type = this.speed.type;
218         c.option = this.option;
219         c._localScope = walker._localScope;
220         return c;
221     };
